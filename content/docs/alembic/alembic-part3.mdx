## 수동 마이그레이션 스크립트 작성

autogenerate가 완벽하지 않거나 감지하지 못하는 복잡한 수동 변경(이름 변경, 인덱스 커스텀)을 파일 튜닝으로 처리합니다.

```python
from alembic import op
import sqlalchemy as sa

def upgrade():
    # 테이블 이름 변경
    op.rename_table('old_users', 'new_users')

    # 신규 컬럼 생성 (NotNull 속성 추가 등)
    op.add_column('new_users', sa.Column('is_active', sa.Boolean(), server_default='true'))

    # 기존 컬럼 타입 변경 (Integers -> String)
    op.alter_column('new_users', 'age', type_=sa.String(10))

def downgrade():
    # 롤백 로직은 항상 upgrade의 완벽한 역순 대응으로 작성되어야 함
    op.alter_column('new_users', 'age', type_=sa.Integer())
    op.drop_column('new_users', 'is_active')
    op.rename_table('new_users', 'old_users')
```

## 데이터 마이그레이션 병행 (Data Migration)

단순 테이블 구조 변경뿐만 아니라, 특정 값들을 일괄 변환하거나 새로운 컬럼에 초기 데이터를 부여하고 싶을 때 사용합니다.

```python
from alembic import op
import sqlalchemy as sa

def upgrade():
    # 1. 새 컬럼을 추가 (nullable 허용상태로 먼저 추가)
    op.add_column('users', sa.Column('status', sa.String(20)))

    # 2. op.execute() 로 순수 SQL 실행 -> 기존 데이터들을 일괄 채워넣음
    op.execute("UPDATE users SET status = 'active' WHERE status IS NULL")

    # 3. 데이터가 모두 찼으므로, 제약을 더 엄격히(NOT NULL) 변경
    op.alter_column('users', 'status', nullable=False)

def downgrade():
    op.drop_column('users', 'status')
```

## 제약 우회 (SQLite Batch Mode)

SQLite는 컬럼 변경이나 삭제(Drop) 지원이 부족합니다. 이를 우회하기 위해 임시 테이블 활용 체계인 Batch 모드를 씁니다.

```python
from alembic import op
import sqlalchemy as sa

def upgrade():
    # SQLite 환경을 염두에 둔다면, 컬럼 삭제는 반드시 batch_alter_table 컨텍스트에서 수행해야 합니다.
    with op.batch_alter_table('users', schema=None) as batch_op:
        # 임시 테이블 생성 후 이전 등의 마술로 SQLite의 한계를 극복함
        batch_op.drop_column('old_column')

def downgrade():
    with op.batch_alter_table('users', schema=None) as batch_op:
        batch_op.add_column(sa.Column('old_column', sa.String(50)))
```
