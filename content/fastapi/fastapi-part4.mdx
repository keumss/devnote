## 입력·수정·공개 모델과 저장소 분리하기

클라이언트 입력, 부분 수정, 공개 응답의 필드는 서로 다릅니다. 서버가 정하는 ID·작성자·생성 시각은 입력 모델에서 제외합니다.

```python
# app/models.py
from datetime import datetime

from pydantic import BaseModel, ConfigDict, Field


class PostCreate(BaseModel):
    title: str = Field(min_length=1, max_length=100)
    content: str = Field(min_length=1, max_length=10_000)


class PostUpdate(BaseModel):
    title: str | None = Field(default=None, min_length=1, max_length=100)
    content: str | None = Field(default=None, min_length=1, max_length=10_000)


class PostPublic(BaseModel):
    model_config = ConfigDict(from_attributes=True)
    id: int
    title: str
    content: str
    author: str
    created_at: datetime
```

학습용 저장소는 서버 재시작 시 사라지고 여러 워커 사이에 공유되지 않습니다. 운영에서는 같은 메서드 계약을 SQLModel 저장소로 교체하고 DB UNIQUE 제약과 트랜잭션을 사용합니다.

```python
# app/repository.py
from dataclasses import dataclass
from datetime import datetime, timezone

from app.models import PostCreate


@dataclass
class PostRecord:
    id: int
    title: str
    content: str
    author: str
    created_at: datetime


class DuplicateTitleError(Exception):
    pass


class PostRepository:
    def __init__(self) -> None:
        self._posts: dict[int, PostRecord] = {}
        self._next_id = 1

    def title_exists(self, title: str, exclude_id: int | None = None) -> bool:
        return any(
            item.id != exclude_id and item.title.casefold() == title.casefold()
            for item in self._posts.values()
        )

    def add(self, data: PostCreate, author: str) -> PostRecord:
        if self.title_exists(data.title):
            raise DuplicateTitleError(data.title)
        post = PostRecord(
            self._next_id,
            data.title,
            data.content,
            author,
            datetime.now(timezone.utc),
        )
        self._posts[post.id] = post
        self._next_id += 1
        return post

    def get(self, post_id: int) -> PostRecord | None:
        return self._posts.get(post_id)

    def all(self) -> list[PostRecord]:
        return sorted(self._posts.values(), key=lambda item: item.id)

    def delete(self, post_id: int) -> None:
        del self._posts[post_id]
```

## 생성·목록·검색·상세 조회 구현하기

저장소를 의존성으로 제공하면 테스트에서 교체할 수 있습니다. 고정된 `/search`는 `/{post_id}`보다 먼저 등록합니다.

```python
# app/api/posts.py
from typing import Annotated

from fastapi import APIRouter, Depends, HTTPException, Query, status

from app.models import PostCreate, PostPublic, PostUpdate
from app.repository import DuplicateTitleError, PostRecord, PostRepository

router = APIRouter(prefix="/posts", tags=["posts"])
repository = PostRepository()


def get_repository() -> PostRepository:
    return repository


Repo = Annotated[PostRepository, Depends(get_repository)]


def get_post_or_404(post_id: int, repository: PostRepository) -> PostRecord:
    post = repository.get(post_id)
    if post is None:
        raise HTTPException(status_code=404, detail="Post not found")
    return post


@router.post("", response_model=PostPublic, status_code=status.HTTP_201_CREATED)
def create_post(data: PostCreate, repository: Repo) -> PostRecord:
    try:
        return repository.add(data, author="demo-user")
    except DuplicateTitleError as exc:
        raise HTTPException(status_code=409, detail="Title already exists") from exc


@router.get("", response_model=list[PostPublic])
def read_posts(
    repository: Repo,
    offset: Annotated[int, Query(ge=0)] = 0,
    limit: Annotated[int, Query(ge=1, le=100)] = 20,
) -> list[PostRecord]:
    return repository.all()[offset : offset + limit]


@router.get("/search", response_model=list[PostPublic])
def search_posts(
    repository: Repo,
    q: Annotated[str, Query(min_length=1)],
) -> list[PostRecord]:
    keyword = q.casefold()
    return [item for item in repository.all() if keyword in item.title.casefold()]


@router.get("/{post_id}", response_model=PostPublic)
def read_post(post_id: int, repository: Repo) -> PostRecord:
    return get_post_or_404(post_id, repository)
```

생성은 201과 공개 모델을 반환합니다. 검색 결과가 없으면 빈 배열, 특정 ID가 없으면 404를 반환합니다.

## PATCH와 DELETE의 경계 조건 처리하기

`exclude_unset=True`는 실제로 보낸 필드만 남깁니다. 이 API는 명시적인 `null`을 허용하지 않으며 빈 PATCH는 현재 리소스를 그대로 반환합니다.

```python
from fastapi import Response


@router.patch("/{post_id}", response_model=PostPublic)
def update_post(post_id: int, update: PostUpdate, repository: Repo) -> PostRecord:
    post = get_post_or_404(post_id, repository)
    changes = update.model_dump(exclude_unset=True)
    if any(value is None for value in changes.values()):
        raise HTTPException(status_code=422, detail="Fields cannot be null")
    title = changes.get("title")
    if title and repository.title_exists(title, exclude_id=post.id):
        raise HTTPException(status_code=409, detail="Title already exists")
    for field, value in changes.items():
        setattr(post, field, value)
    return post


@router.delete("/{post_id}", status_code=status.HTTP_204_NO_CONTENT)
def delete_post(post_id: int, repository: Repo) -> Response:
    get_post_or_404(post_id, repository)
    repository.delete(post_id)
    return Response(status_code=status.HTTP_204_NO_CONTENT)
```

빈 문자열은 모델의 `min_length`에서 422가 됩니다. 성공한 DELETE는 본문 없는 204를 반환합니다.

## 404·409·422 오류의 책임 구분하기

```text
404  요청한 게시물이 없음
409  현재 저장 상태와 제목이 충돌함
422  경로·쿼리·본문 검증 규칙을 위반함
```

잘못된 ID 타입과 페이지 범위는 FastAPI가 경로 실행 전에 422로 처리합니다. 저장소 오류는 라우터가 HTTP 상태로 바꿉니다. 저장 기술을 교체해도 이 공개 계약은 유지합니다.

## TestClient와 dependency_overrides로 격리하기

`TestClient`는 실제 포트를 열지 않습니다. context manager는 lifespan을 실행하며, 테스트마다 새 저장소를 주입하면 실행 순서에 의존하지 않습니다.

```python
# tests/conftest.py
from collections.abc import Iterator

import pytest
from fastapi.testclient import TestClient

from app.api.posts import get_repository
from app.main import app
from app.repository import PostRepository


@pytest.fixture
def client() -> Iterator[TestClient]:
    test_repository = PostRepository()
    app.dependency_overrides[get_repository] = lambda: test_repository
    try:
        with TestClient(app) as test_client:
            yield test_client
    finally:
        app.dependency_overrides.clear()
```

override 키는 라우터가 사용한 원래 의존성 함수 객체여야 합니다. 테스트가 끝나면 반드시 지워 다음 테스트로 상태가 새지 않게 합니다.

## 정상·검증·충돌·누락 시나리오 테스트하기

내부 구현보다 상태 코드와 공개 JSON을 검사합니다.

```python
def test_post_contract(client) -> None:
    created = client.post(
        "/posts",
        json={"title": "테스트", "content": "계약 확인"},
    )
    assert created.status_code == 201
    post_id = created.json()["id"]

    assert client.get("/posts/search", params={"q": "테스트"}).status_code == 200
    assert client.patch(f"/posts/{post_id}", json={"title": None}).status_code == 422
    assert client.get("/posts", params={"limit": 101}).status_code == 422
    assert client.delete(f"/posts/{post_id}").status_code == 204
    assert client.get(f"/posts/{post_id}").status_code == 404
```

같은 제목을 두 번 생성했을 때 409도 추가합니다. Part 5에서는 401·403, Part 6에서는 업로드 실패 후 정리를 이 테스트 흐름에 연결합니다.
