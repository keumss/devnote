이번에는 JSON 파일에서 상품 재고를 읽어 예약 수량만큼 차감하는 CLI를 pytest로 검증합니다. 목표는 정상 예약, 수량 경계, 파일 저장, 환경 변수 경로, 사용자 출력과 종료 상태를 실제 작업 폴더에 영향을 주지 않고 확인하는 것입니다. 첫 토픽의 `inventory.py`를 준비한 뒤 나머지 코드를 `test_inventory.py`에 순서대로 추가합니다.

## inventory.py로 재고 예약 규칙과 JSON 저장 준비하기

테스트할 대상은 계산 규칙, 파일 경계, CLI 경계를 함수로 나눕니다. 기본 파일은 `inventory.json`이지만 `INVENTORY_FILE` 환경 변수로 실행할 파일을 바꿀 수 있게 합니다.

```python
# inventory.py
from __future__ import annotations

import argparse
import json
import os
import sys
from pathlib import Path


def calculate_remaining(stock: int, quantity: int) -> int:
    if quantity <= 0:
        raise ValueError('예약 수량은 1 이상이어야 합니다.')
    if quantity > stock:
        raise ValueError('재고가 부족합니다.')
    return stock - quantity


def load_inventory(path: Path) -> dict[str, int]:
    data = json.loads(path.read_text(encoding='utf-8'))
    if not isinstance(data, dict) or any(
        not isinstance(name, str)
        or not isinstance(stock, int)
        or isinstance(stock, bool)
        or stock < 0
        for name, stock in data.items()
    ):
        raise ValueError('재고 파일 형식이 올바르지 않습니다.')
    return data


def save_inventory(path: Path, inventory: dict[str, int]) -> None:
    path.write_text(
        json.dumps(inventory, ensure_ascii=False, indent=2) + '\n',
        encoding='utf-8',
    )


def reserve(path: Path, name: str, quantity: int) -> int:
    inventory = load_inventory(path)
    if name not in inventory:
        raise ValueError(f'상품을 찾을 수 없습니다: {name}')

    remaining = calculate_remaining(inventory[name], quantity)
    inventory[name] = remaining
    save_inventory(path, inventory)
    return remaining


def main(argv: list[str] | None = None) -> int:
    parser = argparse.ArgumentParser(description='상품 재고 예약')
    parser.add_argument('name')
    parser.add_argument('quantity', type=int)
    args = parser.parse_args(argv)
    inventory_path = Path(os.getenv('INVENTORY_FILE', 'inventory.json'))

    try:
        remaining = reserve(inventory_path, args.name, args.quantity)
    except (FileNotFoundError, json.JSONDecodeError, ValueError) as error:
        print(f'inventory: {error}', file=sys.stderr)
        return 1

    print(f'{args.name}: {remaining}개 남음')
    return 0


if __name__ == '__main__':
    raise SystemExit(main())
```

계산 함수는 파일을 모르고, `reserve()`는 데이터를 읽고 저장하며, `main()`은 인수·환경 변수·출력과 종료 상태를 담당합니다. 이 경계가 뒤의 테스트 범위와 직접 대응합니다.

## parametrize로 예약 수량 경계 검증하기

같은 재고에서 허용되는 수량과 거부되는 수량을 한 표로 확인합니다. 정상 결과와 오류 메시지를 나눠 두 테스트로 작성하면 각 테스트의 성공 조건이 하나로 유지됩니다.

```python
# test_inventory.py
import json
from pathlib import Path

import pytest

import inventory


@pytest.mark.parametrize(
    ('stock', 'quantity', 'expected'),
    [
        pytest.param(5, 1, 4, id='minimum'),
        pytest.param(5, 5, 0, id='all-stock'),
    ],
)
def test_calculate_remaining(stock: int, quantity: int, expected: int) -> None:
    assert inventory.calculate_remaining(stock, quantity) == expected


@pytest.mark.parametrize(
    ('quantity', 'message'),
    [
        pytest.param(0, '예약 수량은 1 이상', id='zero'),
        pytest.param(-1, '예약 수량은 1 이상', id='negative'),
        pytest.param(6, '재고가 부족', id='over-stock'),
    ],
)
def test_calculate_remaining_rejects_invalid_quantity(
    quantity: int,
    message: str,
) -> None:
    with pytest.raises(ValueError, match=message):
        inventory.calculate_remaining(stock=5, quantity=quantity)
```

매개변수 ID가 `minimum`, `zero`, `over-stock`처럼 경계의 의미를 보여 주므로 실패한 입력을 테스트 출력에서 바로 찾을 수 있습니다.

## tmp_path로 재고 파일 변경 격리하기

예약 뒤 JSON이 실제로 갱신되는지 확인하되 프로젝트의 `inventory.json`은 건드리지 않습니다. `tmp_path` 안에 테스트 전용 파일을 만들고 반환값과 저장 결과를 함께 검증합니다.

```python
def write_inventory(path: Path, data: dict[str, int]) -> None:
    path.write_text(json.dumps(data), encoding='utf-8')


def test_reserve_updates_only_selected_product(tmp_path: Path) -> None:
    inventory_path = tmp_path / 'inventory.json'
    write_inventory(inventory_path, {'keyboard': 5, 'mouse': 2})

    remaining = inventory.reserve(inventory_path, 'keyboard', 3)

    assert remaining == 2
    assert json.loads(inventory_path.read_text(encoding='utf-8')) == {
        'keyboard': 2,
        'mouse': 2,
    }
```

반환값만 확인하면 저장 누락을 놓치고 파일만 확인하면 호출자가 받는 결과 오류를 놓칠 수 있습니다. 이 테스트는 한 예약 작업의 두 관찰 가능한 결과를 함께 확인합니다.

## monkeypatch로 INVENTORY_FILE 경로 교체하기

CLI가 기본 파일 대신 환경 변수가 가리키는 파일을 사용하는지 검증할 때 테스트 전용 경로를 환경에 넣습니다.

```python
def test_main_uses_inventory_file_from_environment(
    tmp_path: Path,
    monkeypatch,
    capsys,
) -> None:
    inventory_path = tmp_path / 'warehouse.json'
    write_inventory(inventory_path, {'keyboard': 4})
    monkeypatch.setenv('INVENTORY_FILE', str(inventory_path))

    exit_code = inventory.main(['keyboard', '1'])

    assert exit_code == 0
    assert json.loads(inventory_path.read_text(encoding='utf-8')) == {'keyboard': 3}
    assert capsys.readouterr().out == 'keyboard: 3개 남음\n'
```

테스트가 끝나면 환경 변수는 자동 복원됩니다. 코드의 현재 작업 디렉터리를 바꾸지 않아 다른 테스트의 상대 경로 해석에도 영향을 주지 않습니다.

## capsys로 CLI 성공·실패 출력과 종료 코드 확인하기

사용자가 보는 CLI 계약을 마무리하려면 성공은 표준 출력과 상태 0, 실패는 표준 오류와 상태 1로 끝나는지 확인합니다.

```python
def test_main_reports_missing_product(
    tmp_path: Path,
    monkeypatch,
    capsys,
) -> None:
    inventory_path = tmp_path / 'inventory.json'
    write_inventory(inventory_path, {'keyboard': 2})
    monkeypatch.setenv('INVENTORY_FILE', str(inventory_path))

    exit_code = inventory.main(['mouse', '1'])
    captured = capsys.readouterr()

    assert exit_code == 1
    assert captured.out == ''
    assert captured.err == 'inventory: 상품을 찾을 수 없습니다: mouse\n'
    assert json.loads(inventory_path.read_text(encoding='utf-8')) == {'keyboard': 2}
```

실패한 예약에서 파일이 유지되는지도 확인해 오류 메시지만 맞고 데이터가 손상되는 경우를 막습니다. 이후 잠금이나 데이터베이스를 붙인다면 동시에 예약할 때의 원자성을 별도 통합 테스트로 확장합니다.
